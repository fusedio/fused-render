The whole demo.py — one function, plain Python:

        
def main(peak: float, spread: float, bars: int = 28):
    import math
    values = []
    for i in range(bars):
        x = i / (bars - 1) * 100
        values.append(math.exp(-((x - peak)**2) / (2 * spread**2)))
    return {"values": values}   # plain JSON back to the page
